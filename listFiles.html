const express = require('express'); const fs = require('fs'); const path = require('path'); const app = express(); const port = 3000; app.get('/list-mp3-files', (req, res) => { const directoryPath = path.join(__dirname, 'files/aud'); fs.readdir(directoryPath, (err, files) => { if (err) { return res.status(500).send('Unable to scan directory: ' + err); } const mp3Files = files.filter(file => path.extname(file).toLowerCase() === '.mp3'); res.json(mp3Files); }); }); app.use('/files/aud', express.static(path.join(__dirname, 'files/aud'))); app.listen(port, () => { console.log(`Server is running at http://localhost:${port}`); });
